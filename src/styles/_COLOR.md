.color-1 {

:root {
  --color-1: #d3cbe7;
  --color-2: #98a5dc;
  --color-3: #5d82c0;
  --color-4: #426694;
  --color-5: #293761;

/* Linear Gradient */
.linear-gradient {

}

/* Radial Gradient */
.radial-gradient {
  background: radial-gradient(circle, #d3cbe7, #98a5dc, #5d82c0, #426694, #293761);
}
